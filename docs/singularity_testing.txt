
RUN USING SINGULARITY
-------------------------------------------------------------------------------------------------------------------------------
scp -r seqsender-v1.0.0.tar snu3@dvd-sars2seq-dev01.biotech.cdc.gov:~/seqsender

singularity build seqsender-v1.0.0.sif docker-archive://seqsender-v1.0.0.tar

singularity run --writable-tmpfs seqsender-v1.0.0.sif retrieve --sample_file config --outdir sample_template

singularity run --writable-tmpfs seqsender-v1.0.0.sif check --unique_name reina_submissions

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender prep --unique_name reina_submission --config config_files/reina_config.yaml --metadata test_input/test_metadata.tsv --fasta test_input/test_fasta.fasta --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender submit --unique_name reina_submission --config config_files/reina_config.yaml --metadata test_input/test_metadata.tsv --fasta test_input/test_fasta.fasta --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender genbank --unique_name reina_submission --config config_files/reina_config.yaml --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender sra --unique_name reina_submission --config config_files/reina_config.yaml --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender biosample --unique_name reina_submission --config config_files/reina_config.yaml --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender biosample_sra --unique_name reina_submission --config config_files/reina_config.yaml --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender gisaid --unique_name reina_submission --config config_files/reina_config.yaml --test

singularity run --writable-tmpfs seqsender-v1.0.0.sif seqsender update_submissions
